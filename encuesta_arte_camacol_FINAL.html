// Apps Script v4 — CAMACOL // Ejecutar como: Yo | Acceso: Cualquier usuario const SHEET_NAME = 'Votos'; const HEADERS = ['Marca de tiempo', 'Nombre', 'Obra 1', 'Obra 2', 'Obra 3', 'Obra 4']; function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sh = ss.getSheetByName(SHEET_NAME); if (!sh) { sh = ss.insertSheet(SHEET_NAME); sh.appendRow(HEADERS); const h = sh.getRange(1, 1, 1, HEADERS.length); h.setFontWeight('bold'); h.setBackground('#b8960c'); h.setFontColor('#ffffff'); sh.setFrozenRows(1); } return sh; } // GET handles both reading votes AND writing votes (via query params) // This avoids all CORS issues — everything goes through GET function doGet(e) { const params = e.parameter; // If 'action=vote' is in the URL, save the vote if (params.action === 'vote') { const lock = LockService.getScriptLock(); lock.tryLock(10000); try { const sh = getSheet(); const now = new Date().toLocaleString('es-CO', { timeZone: 'America/Bogota' }); const s = params.selections ? params.selections.split(',') : []; sh.appendRow([now, params.name || '', s[0]||'', s[1]||'', s[2]||'', s[3]||'']); return respond({ status: 'ok' }, params.cb); } catch(err) { return respond({ status: 'error', message: err.toString() }, params.cb); } finally { lock.releaseLock(); } } // Otherwise return current votes try { const sh = getSheet(); const last = sh.getLastRow(); let votes = []; if (last >= 2) { const rows = sh.getRange(2, 1, last - 1, 6).getValues(); votes = rows.filter(r => r[1]).map(r => ({ timestamp: String(r[0]), name: r[1], selections: [r[2],r[3],r[4],r[5]].filter(Boolean) })); } return respond({ status: 'ok', votes }, params.cb); } catch(err) { return respond({ status: 'error', message: err.toString() }, params.cb); } } function respond(data, callback) { const json = JSON.stringify(data); // Support JSONP for older browsers, plain JSON otherwise const body = callback ? `${callback}(${json})` : json; const mime = callback ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON; return ContentService.createTextOutput(body).setMimeType(mime); } // Keep doPost as fallback function doPost(e) { const lock = LockService.getScriptLock(); lock.tryLock(10000); try { const data = JSON.parse(e.postData.contents); const sh = getSheet(); const now = new Date().toLocaleString('es-CO', { timeZone: 'America/Bogota' }); sh.appendRow([now, data.name||'', ...(data.selections||[]).slice(0,4).concat(['','','','']).slice(0,4)]); return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:'error',message:err.toString()})).setMimeType(ContentService.MimeType.JSON); } finally { lock.releaseLock(); } }
